1function [ Thrust,B_ref ] = Position_Controller( e1_outer, x2, psi_des, dt, kp_outer, kv_outer, G_outer, m, Adaptive_on )
 2
 3global thetaf_est_outer thetaf_est_dot_outer theta_est_outer_int Wf_outer e2f_outer
 4 
 5B_outer = kp_outer + kv_outer;
 6
 7%calculate theta and theta_integrated
 8theta_est_outer=thetaf_est_dot_outer+B_outer*thetaf_est_outer;
 9theta_est_outer_int=theta_est_outer_int+theta_est_outer*dt;
10
11%calculate the desired acceleration and velocity
12acc_des=-theta_est_outer;
13vel_des=-theta_est_outer_int;
14
15%calculate the errors
16e2_outer=x2-vel_des;%error in velocity
17W_outer=[0;0;9.81];
18
19%calculate the derivatives of filter variables
20Wf_dot_outer=-B_outer*Wf_outer+W_outer;
21e2f_dot_outer=-B_outer*e2f_outer+e2_outer;
22
23thetaf_est_dot_outer=Adaptive_on*G_outer*((e2f_outer+e1_outer));
24
25% Calculate the control (u_outer)
26uf_outer=m*(Wf_outer-kp_outer*e1_outer-kv_outer*e2f_outer-thetaf_est_outer);
27uf_dot_outer=m*(Wf_dot_outer-kp_outer*e2_outer-kv_outer*e2f_dot_outer-thetaf_est_dot_outer);
28u_outer=uf_dot_outer+B_outer*uf_outer;
29
30Thrust=((u_outer(1,1)^2)+(u_outer(2,1)^2)+(u_outer(3,1)^2))^0.5;%this is the desired thrust in body frame
31
32%Calculate the desired/reference attitude quaternion (B_ref) to be sent to the Attitude controller (inner loop)
33Fb_cap=[0;0;1];
34Fi_cap=u_outer/norm(u_outer);
35B_ref_p=(1/((2*(1+dot(Fb_cap',Fi_cap')))^0.5))*[(1+dot(Fb_cap',Fi_cap'));-cross(Fb_cap',Fi_cap')'];
36B_ref=quatmultiply(B_ref_p',[cos(psi_des/2);0;0;sin(psi_des/2)]')';
37B_ref=B_ref./norm(B_ref);
38
39%Filter variables update
40Wf_outer=Wf_outer+Wf_dot_outer*dt;
41e2f_outer=e2f_outer+e2f_dot_outer*dt;
42thetaf_est_outer=thetaf_est_outer+thetaf_est_dot_outer*dt;
43
44
45end
